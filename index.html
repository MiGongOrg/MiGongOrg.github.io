JavaScript eval() 安全隐患

Posted on 2016-06-15   |   In JavaScript

为什么 eval() 存在安全隐患

eval() 函数可以将字符进行编译后执行，如果使用 eval() 处理 JSON 数据，当服务器本身或发来的 JSON 数据被劫持，就很容易被利用，在客户端编译执行恶意代码。

如何防范和避免

  1. 使用 JSON.parse() 代替 eval() 因为 JSON.parse() 更加安全
  2. 将 JSON 数据中的 html 字符进行转码

Backbone 集合（collection）中提到的 underscore 方法

Posted on 2016-05-06   |   In JavaScript / JQuery

each 遍历（传入一个对象或数组）

var oEach = {name:'Ming', age:12}
_.each(oEach,function(value, index){
  console.log([index] + ':' + [value]); //name:Ming   age:12
});

map 遍历 （传入一个对象或数组）

var map = [1, 2, 3, 4, 8, 6, 5];
var mapObj = {one:1, Two:2, Three:3};

//将数组中的每个值*3
_.map(map, function(number){
  console.log(number * 3);  //3,6,9,15,24,18,15
});

//将每个对象的值遍历*3
_.map(mapObj, function(index,value){
  console.log(value + ':' + index * 3); //one:3 two:6 three:9
});

reduce 将数组内数字相加求和

var reduces = [5, 6, 10];
var oX = function(one, two) {
  return one + two;
};
console.info(_.reduce(reduces, oX));    //21
console.info(_.reduce(reduces, oX, 2)); //23

//reduceRight 从右侧开始组合元素

var oReduceRight = [[2, 5], [9, 6], [3, 1]];
var oReduceRightFn = function(a, b) {
  return a.concat(b);
};
console.log(_.reduceRight(oReduceRight, oReduceRightFn, [])); //[3, 1, 9, 6, 2, 5]

find 遍历查找（遍历，查找满足条件的值，找到匹配的元素，函数将立即返回，不会遍历整个list）

var list = [1, 2, 3, 4];
_.find(list, function(nb){
  if (nb % 2 == 0) {
    return console.log(nb); //2（满足条件后立即返回，所以不会遍历到4）
  }
});

filter 遍历 list 中的每个值，返回满足条件检测到的值

var list = [1, 2, 3, 4];
_.filter(list, function(nb){
  if (nb % 2 == 0) {
    return console.log(nb); //2 和 4（偶数都满足条件）
  }
});

reject 与 filter 相反，遍历 list 中的每个值，返回不满足条件检测到的值

var list = [1, 2, 3, 4];
_.reject(list, function(nb){
  if (nb % 2 == 0) {
    return console.log(nb); //1 和 3（reject 与 filter 相反）
  }
});

every 当数组中的所有值为 true 时，则返回 true。否则返回 false

var oEvery = [true, 1, null, 'yes'];
console.info(_.every(oEvery));  //false

some 当数组中某个值满足 true 则返回 true，否则则返回 false。（其中 ‘yes’ 是字符串，是 true）

var oSome = [null, 0, 'yes', false];
console.log(_.some(oSome)); //true

contains 数组中是否包含该值

var oContains = [1, 2, 3, 4];
console.log(_.contains(oContains, 3));    //true(因为包含 3 所以返回 true)
console.info(_.contains(oContains, 8));   //false(不包含 8 所以返回 false)

invoke 将数组中的数组进行排序

var oInvoke = [[1, 5, 3,11],[9, 8, 7]];
console.log(_.invoke(oInvoke, 'sort')); //[[1, 11, 3, 5], [7, 8, 9]]

max 遍历返回最大值

var oMax = [{name:'tom',age:12},{name:'lili',age:11},{name:'going',age:99}];
var oMaxfn = function(e){
  return e.age;
};
console.info(_.max(oMax, oMaxfn));  //age 最大值的那个对象 {name:'going',age:99}

min 遍历返回最小值

var oMin = [2, 6, 9];
console.log(_.min(oMin)); //2(最小值是2)

sortBy 使用某种方式进行排序

var oSortBy = [1, 3, 5, 7, 9];
var oSortByObj = [{name:'c',age:7},{name:'a',age:6},{name:'b',age:5}];
var oSortByFn = function(number){
  return Math.sin(number);
};
console.info(_.sortBy(oSortBy, oSortByFn)); //5 3 9 7 1
console.log(_.sortBy(oSortByObj, 'name'));  //a b c（按照对象值进行排序）

groupBy 将数组中的值，按照指定方式进行分类

var oGroupBy = ['one','two','three'];
var oGroupByNb = [1.2, 1.5, 2, 2.3];
var oGroupByNbFn = function(number) {
  return Math.floor(number);
};
console.log(_.groupBy(oGroupBy,'length'));          //3: ['one','two'] 5: ['three']
console.info(_.groupBy(oGroupByNb, oGroupByNbFn));  //1: [1.2, 1.5] 2: [2, 2.3]

indexBy 通过键值返回对象排序

var oIndexBy = [{name:'c',age:7},{name:'a',age:6},{name:'b',age:5}];
console.log(_.indexBy(oIndexBy, 'age'));  // {'5':{name:'b',age:5},'6':{name:'a',age:6},'7':{name:'c',age:7}}

countBy 返回符合条件值的数量（计算数量）

var oCountBy = [1, 2, 3, 4, 5, 6, 7];
var countFn = function(e) {
  return e % 2 == 0 ? 'even' : 'odd';
};
console.log(_.countBy(oCountBy, countFn));  //{odd:4, even:3}

shuffle 随机打乱数组排序

var oShuffle = [1, 2, 3, 4, 5, 6];
console.log(_.shuffle(oShuffle)); //每次刷新随机打乱排序

sample 从数组中随机获取数值或指定随机获取[N]个值

var oSample = [1, 2, 3, 4, 5, 6];
console.log(_.sample(oSample));     //从数组中随机获取一个数值
console.info(_.sample(oSample, 4)); //从数组中随机获取4个数值生成新数组

toArray 将对象转换为数组

var oToArray = {a:0,b:1,c:2,d:3};
console.log(_.toArray(oToArray)); //[0, 1, 2, 3]

size 获得数组或对象的 length

var oSzie = {a:0,b:1,c:2,d:3};
var oSzieArray = [1, 2, 3, 4, 5, 6];
console.log(_.size(oSzie));       //4
console.info(_.size(oSzieArray)); //6

partition 拆分一个数组，为两个数组。第一个数组满足自己所定义的条件

var oPartition = [1, 2, 3, 4, 5, 6];
var isOdd = function(e) {
  return e % 2 != 0;
};
console.log(_.partition(oPartition, isOdd));  //[[1, 3, 5], [2, 4, 6]]

underscore 针对 Array 的操作方法

first 返回数组中的第一或前[N]个值

var oFirst = [1, 2, 3, 4, 5, 6];
console.info(_.first(oFirst));    //1
console.log(_.first(oFirst, 3));  //[1, 2, 3]

initial 返回除最后一个的所有数组，或指定返回除[N]个后返回

var oInitial = [1, 2, 3, 4, 5, 6];
console.log(_.initial(oInitial));     //[1, 2, 3, 4, 5]
console.info(_.initial(oInitial, 2)); //[1, 2, 3, 4]

last 返回最后一个数组值或指定返回后[N]个值

var oLast = [1, 2, 3, 4, 5, 6];
console.log(_.last(oLast));     //6
console.info(_.last(oLast, 4)); //[3, 4, 5, 6]

without 删除数组中指定的值，并返回删除后的结果

var oWithout = [1, 2, 3, 4, 5, 6];
console.log(_.without(oWithout, 1, 5)); // [2, 3, 4, 6]

difference 类似于without，但返回的值来自array参数数组，并且不存在于other 数组。

var oDifference = [1, 2, 3, 4, 5, 6];
console.log(_.difference(oDifference,[1,4,6])); //[2, 3, 5]

indexOf 返回 value 在该 array 中的索引值，如果 value 不存在 array 中就返回-1。

var oIndexOf = [1, 2, 3, 4, 5, 6];
console.log(_.indexOf(oIndexOf, 4));  //3

lastIndexOf 返回 value 在该 array 中的从最后开始的索引值，如果value不存在 array 中就返回-1。

var oLastIndexOf = [1, 2, 3, 4, 2, 6];
console.info(_.lastIndexOf(oLastIndexOf, 2)); //4

underscore Obj 对象

isEmpty 判断该对象或数组是否为空，返回 true 或 false

var oIsEmpty = [1, 2, 3, 4, 5, 6];
var oIsEmptyObj = {name:'a',age:12};
console.log(_.isEmpty(oIsEmpty));     //false
console.info(_.isEmpty(oIsEmptyObj)); //false
console.log(_.isEmpty({}));           //true
console.info(_.isEmpty([]));          //true

underscore 链式语法

chain 封装对象

var oChain = [{name:'c',age:7},{name:'a',age:6},{name:'b',age:5}];
var oChainObj = _.chain(oChain);
console.info(oChainObj);  //封装后的对象

//value 获取封装对象中的值

grunt-contrib-less 与 grunt-autoprefixer 配置使用介绍

grunt-contrib-less 可以自动化实现将 LESS 文件编译成 CSS 文件。

安装：grunt-contrib-less

npm install grunt-contrib-less --save-dev

配置：grunt-contrib-less

module.exports = function (grunt) {
  grunt.initConfig({
    less:{
      page: {
        options: {
        //配置项
        compress: true  //CSS 压缩
        },
        files: {
          //目标文件，将 page.less 文件编译压缩后，生成 page.css 文件
          'dist/css/page.css' : 'src/css/page.less'
        }
      }
    }
  });
  //加载任务
  grunt.loadNpmTasks('grunt-contrib-less');
  //执行任务
  grunt.registerTask('pageLess',['less:page']);
};

详细介绍：grunt-contrib-less

grunt-autoprefixer 处理 CSS 属性私有前缀，通过配置实现自动化添加 CSS 属性私有前缀。

安装：grunt-autoprefixer

npm install grunt-autoprefixer --save-dev

配置：grunt-autoprefixer

module.exports = function (grunt) {
  grunt.initConfig({
    autoprefixer: {
      options: {
        //配置项
        browsers: ['last 2 versions', 'ie 8', 'ie 9']  //浏览器兼容
      },
      css: {
        //目标文件
        src: [
          'dist/css/page.css'  //将哪个 CSS 文件中的 CSS 属性添加私有前缀
        ]
      }
    }
  });
  //加载任务
  grunt.loadNpmTasks('grunt-autoprefixer');
  //执行任务
  grunt.registerTask('autoprefixerCss',['autoprefixer:css']);
};

RequireJS 模块化加载 与 uglify 文件压缩

Posted on 2016-04-19   |   In JavaScript / JQuery

关于 RequireJs 它可以用来管理页面中使用的 JS 库之间的依赖关系，同时它也可以让我们以模块化的形式组织 JS 代码。而 uglify 就是将 JS 文件自动化压缩，可以设置混淆变量名，合并 JS 文件等。

安装 RequireJS 与 uglify

RequireJS ：bower install requirejs

uglify ：npm install grunt-contrib-uglify --save-dev

RequireJS 的使用

在适当的位置引用，其中 require.js 是我们通过 Bower 下载下来的 RequireJS，而 data-main 中定义的则是 RequireJS 的配置文件，需要我们手动新建和修改（config.js）,在引用时不需要添加.js后缀。

RequireJS 的 config 配置文件

require.config({
  //paths 模块声明（不需要添加.js后缀）
  paths: {
    'test' : 'test',  //test 是与 config.js 同级目录下的 test.js 文件
    'jquery': '../../assets/jquery/js/jquery.min'
  },
  shim: {
    //处理一些没有遵守 AMD 规范的 js 库
    'underscore':{
      exports: '_'  //exports值（输出的变量名），表明这个模块外部调用时的名称
    },
    'backbone': {
      deps: ['underscore', 'jquery'], //deps数组，表明该模块的依赖性
      exports: 'backbone'
    }
  }
});

//加载相应 JS 文件
require(['test','jquery'],function(t) {
  t.hello();  //执行回调函数
});

test.js 文件中 RequireJs 模块式的写法

define([
  'jquery'  //引用 jQuery
  ], function($) {
  return {
    hello: function(){
      console.log('Hello World');
      $('.container').hide();
    }
  };
});

uglify 文件压缩的使用

通过 Gruntfile.js 配置文件来添加 uglify 文件压缩任务

module.exports = function (grunt) {
  // 项目配置
  grunt.initConfig({
    //压缩文件
    uglify:{
      // 加注释
      options:{
        banner:"/* <%= pkg.name="" %=""> <%= pkg.version="" %="">*/ \n"
      },
      buildOne:{
        files:{
          //将 config.js 压缩成 config.min.js
          'dist/js/config.min.js' : 'src/js/config.js'
        }
      },
      buildAll:{
        //将某个文件夹下的所有JS文件按结构压缩到指定的文件夹下
        files:[{
          expand: true,
          cwd:'src/js',   //JS目录下
          src:'**/*.js',  //所有JS文件
          dest:'dist/js'  //压缩到某个目录
        }]
      }
    }
  });
  //加载
  grunt.loadNpmTasks('grunt-contrib-uglify');  //文件压缩

  //执行任务
  grunt.registerTask('buOne', ['uglify:buildOne']);  //grunt buOnt
  grunt.registerTask('buAll', ['uglify:buildAll']);  //grunt buAll
};

Grunt 和 Bower 自动化管理项目依赖库

Posted on 2016-04-19   |   In JavaScript / JQuery

在初探 Grunt 自动化构建工具中已经简单的介绍 Grunt 的使用，其中也提到了 Bower。其实 Bower 和 npm 从某种意义上来说很相似，Bower 可以用来管理常用的 JS 库，比如：jQuery、Backbone 等等。

全局安装 Bower ：npm install bower -g

初始化 Bower ：bower init

更新某个库 ：bower update jquery

搜索关键词 ：bower search jquery

查看 Bower 当前的版本 ：bower -v

初始化 Bower 可以根据提示创建 bower.json 配置文件，类似于 npm init 生成的 package.json 配置文件。

Bower 安装成功后，通过它下载项目中所需要的 JS 库

bower install jquery --save
bower install backbone --save

之后你会发现在项目的根目录下生成了一个名为 bower_components 的文件夹，这里存放的是通过 Bower 下载下来的 JS 项目库。细心的你肯定会发现，它还会识别 JS 库之间的依赖关系，当你下载 Backbone 的时候 Bower 会自动帮你下载好 underscore。

通过 Bower 下载的 JS 库都是整个项目的，而我们其实只需要这个项目的 1-2 个 JS 文件即可，要实现自动化就不能人工去查找提取，所以就有了 grunt-bower-task

通过 npm 安装 grunt-bower-task

安装 grunt-bower-task ：npm install grunt-bower-task --save-dev

之后需要在 Gruntfile.js 配置 grunt-bower-task 达到自动提取 JS 库文件到指定目录下的目的。

module.exports = function (grunt) {
  // 项目配置
  grunt.initConfig({
    //bower
    bower:{
      install:{
        options:{
          targetDir: './assets', //cmd 运行 grunt 或 grunt bower 时，会将JS项目库文件提取到指定的目录下
          layout: 'byComponent'  //设置 byType 或 byComponent 会改变目录结构
        }
      }
    }
  });
  //加载插件
  grunt.loadNpmTasks('grunt-bower-task');

  //默认执行任务(数组中依次执行默认任务)
  grunt.registerTask('default', ['bower:install']);
};

当在 cmd 下运行 grunt 或 grunt bower 时，会将JS项目库文件提取到指定的目录下。在这之前你还需要配置一下 bower.json 文件，明确需要提取哪些文件和要将提取出来的文件放到哪里。

设置 layout 的 Value：byType 或 byComponent 会改变目录结构，grunt-bower-task 的相关配置请参考：点这里

{
  "exportsOverride": {
    "bootstrap": {
      "js": "js/*.js",
      "css": "dist/css/*.css",
      "fonts": "dist/fonts/*.*"
    },
    "backbone": {
      "js": "*min.js"
    },
    "underscore": {
      "js": "*min.js"
    },
    "jquery": {
      "js": "dist/jquery.min.js"
    }
  }
}

如何删除 npm 和 Bower 下载安装的模块？

npm ：npm uninstall grunt-bower-task

Bower ：bower uninstall jquery

如果你要删除 underscore 它还会提示你：它与 Backbone 之间是有依赖关系的，是否继续删除？这样会避免一些有依赖关系的库误删的情况。

初探 Grunt 自动化构建工具

Posted on 2016-04-18   |   In JavaScript / JQuery

如果你和我一样是 Win 开发环境，在安装 Grunt 之前，建议先安装 Cmder，不是必须！

Cmder：http://cmder.net/

Cmder Github：https://github.com/cmderdev/cmder

Grunt 通过 npm 安装和管理，npm 是 Node.js 的包管理器。首先需要安装 Node.js ,快捷键 Win+R 运行 Cmder/cmd 通过命令行安装 Grunt。

安装、检测与更新

安装 Node.js 新版本已经集成了 npm ,所以只需要检测一下版本看是否安装成功

检测 Node.js 版本：node -v

检测 NPM 版本：npm -v

升级更新 NPM ：npm update -g

全局安装 Grunt ：npm install -g grunt

新建 Grunt 项目时，需要在该项目的根目录下新建 package.json 和 Gruntfile.js 配置文件

package.json 配置文件

通过 Cmder 命令行 npm init 按提示步骤创建文件内容。该文件可手动修改，或在下载插件时也会自动更新。

安装 Grunt ：npm install grunt --save 之后查看 package.json 文件，你会发现该文件已被，新增了

"dependencies": {
  "grunt": "^1.0.1"
}

项目的根目录下也会新增一个 node_modules 文件夹，在以后安装的插件都会下载到这个文件夹下。

接下来我们尝试安装 Bower npm install -g bower ,安装成功之后，你会发现 package.json 文件中新增了

"dependencies": {
  "bower": "^1.7.9",
  "grunt": "^1.0.1"
}

在查看 node_modules 文件夹，你会发现也新增了一个 bower 文件夹，相信这样就能很清楚的明白它的工作原理。

一个简单的 package.json 配置文件模板

{
  "name": "my-project-name",
  "version": "0.1.0",
  "devDependencies": {
    "grunt": "^0.4.5",
    "grunt-contrib-jshint": "^0.10.0",
    "grunt-contrib-nodeunit": "^0.4.1",
    "grunt-contrib-uglify": "^0.5.0"
  }
}
**Gruntfile.js 配置文件** 在 Gruntfile.js 配置文件中要执行的则是 grunt 插件执行的相应任务，uglify 压缩文件插件，一个简单的 Gruntfile.js 配置文件例子。 
module.exports = function (grunt) {
  // 项目配置
  grunt.initConfig({
    //读取配置项
    pkg:grunt.file.readJSON("package.json"),
    //执行的具体任务

    //压缩文件
    uglify:{
      // 加注释
      options:{
        banner:"/* <%= pkg.name %> <%= pkg.version %>*/ \n"
      },
      build:{
        //被压缩文件的路径
        src: "js/jquery.js",
        //被压缩后的文件路径
        dest: "dest/jquery.min.js"
      }
    }
  });
  //加载插件
  grunt.loadNpmTasks("grunt-contrib-uglify");
  //默认执行任务(数组中依次执行默认任务)
  grunt.registerTask("default",["uglify"]);
};

给 Grunt 指定任务，当在项目下运行 Cmder 执行 grunt 时，每个任务都会按照其出现的指定顺序依次执行。

grunt.registerTask(taskName, [description, ] taskList)

定义一个要执行的默认任务

grunt.registerTask('default', ['jshint', 'qunit', 'concat', 'uglify']);

还可以给任务指定参数，别名 “assets” 将执行 “cssmin” 和 “uglify” 两个任务，并且它们都带有一个 “forBower” 参数

grunt.registerTask('assets', ['uglify:forBower', 'cssmin:forBower']);

CSS animation-timing-function 属性中的 steps() 与 step-start，step-end

Posted on 2016-01-29   |   In HTML / CSS

steps() 设置间隔参数，可以实现分步过渡

第一个参数指定了时间函数中的间隔数量（必须是正整数）

第二个参数可选，接受 start 和 end 两个值，指定在每个间隔的起点或是终点发生阶跃变化，默认为 end。

steps() 的实现方法：

.heart{
    background-image: url('images/heart-sprite.png');
    -webkit-animation: animate 1s steps(28) infinite;
            animation: animate 1s steps(28) infinite;
}
.star{
    background-image: url('images/star-sprite.png');
    -webkit-animation: animate 1s steps(28) infinite;
            animation: animate 1s steps(28) infinite;
}
@keyframes animate {
    from {
        background-position: 0 0;
    }
    to {
        background-position: -2800px 0;
    }
}

step-start 可以实现与 steps() 效果相同的动画

step-start 等同于 steps(10,start) 动画分成10步，动画执行时为开始左侧端点的部分为开始。

step-end 等同于 steps(10,end) 动画分成10步，动画执行时以结尾端点为开始，默认值为 end。
