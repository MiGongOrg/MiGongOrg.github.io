JavaScript 判断移动端横屏或竖屏

Posted on 2016-01-12   |   In JavaScript / JQuery

JavaScript 判断移动端横屏或竖屏 Code :

通过监听 window.onorientationchange 是否有改变来判断移动端设备横屏或竖屏，其中 window.orientation 可以判断相应角度，如：

竖屏：window.orientation == 180 || window.orientation == 0
横屏：window.orientation == 90 || window.orientation == -90

window.onorientationchange = function(){
    if(window.orientation == 0){    //简单的判断方法
        alert("竖屏");
    } else {
        alert("横屏");
    }
};

如果部分移动端设备不支持 onorientationchange 事件，也可以使用 resize

var evt = "onorientationchange" in window ? "orientationchange" : "resize";
window.addEventListener(evt, function() {
    alert("屏幕旋转");
}, false);

解读 Canvas 动画之 JavaScript

Posted on 2015-11-05   |   In JavaScript / JQuery

日常工作中所接触到的 Canvas 动画比较少，今日偶然在 Codepen 上发现了一个很有趣，且立体感十足的 Canvas 动画，所以借此机会将代码注释并学习一下。

Canvas 动画代码来源：

Author Page: http://codepen.io/enxaneta/

Works Code: http://codepen.io/enxaneta/pen/QjVjWj

Inspired By: http://codepen.io/thebabydino/pen/DKaLo/

为移动端可呈现，以下代码做了简单修改。（部分效果也可简单修改再次扩展，如对该 Canvas 动画有兴趣可查看作者 Codepen 中的 Works Code）

Canvas Animation JavaScript Code：

/* Ming 注释 & 学习 2015.11.05 */
var c = document.getElementById("c");
var ctx = c.getContext("2d");

var cw = c.width = 300,
  cx = cw / 2;
var ch = c.height = 300,
  cy = ch / 2;
var rad = Math.PI / 180;    //Math.PI （3.1415926）

ctx.shadowBlur = 5; //阴影模糊 5
ctx.shadowOffsetX = 0;  //阴影水平偏移量
ctx.shadowOffsetY = 0;  //阴影垂直偏移量
ctx.shadowColor = "#fff";   //阴影颜色 
ctx.lineWidth = 2;  //线宽

var nCusped = document.getElementById("Cusped").value;  //获取 range 改变值
var R = 100,
  r = R / nCusped,
  x, y, t;
var t = 0;

/* 绘制 */
function Draw() {
  t += .5;
  phi = t * rad;    //t * rad（Math.PI / 180）
  x = cx + (R - r) * Math.cos(phi) + r * Math.cos((R - r) / r * phi);   //Math.cos(x) x余弦值。返回的是 -1.0 到 1.0 之间的数；
  y = cy + (R - r) * Math.sin(phi) - r * Math.sin((R - r) / r * phi);   //Math.sin(x) x正玄值。返回值在 -1.0 到 1.0 之间；
  strokeSquare(x, y, 50, phi, "hsl(" + t + " ,99%, 60%)");

  requestId = window.requestAnimationFrame(Draw);   //requestAnimationFrame 执行动画 Draw
}

/* 绘制图形（正方形） */
function strokeSquare(cx, cy, r, a, hsl) {
  ctx.strokeStyle = hsl;  //hsl颜色
  ctx.beginPath();  //开始
  for (var i = 0; i < 4; i++) {
    var x = cx + r * Math.cos(i * Math.PI / 2 + a);
    var y = cy + r * Math.sin(i * Math.PI / 2 + a);
    ctx.lineTo(x, y);   //画直线，i < 4（绘制正方形）；如果改成 i < 4 将（绘制三角形）
  }
  ctx.closePath()   //结束
  ctx.stroke(); //绘制
}

/* 开始 */
function start() {
  requestId = window.requestAnimationFrame(Draw);
  stopped = false;
}

/* 停止 */
function stopAnim() {
  if (requestId) {
    window.cancelAnimationFrame(requestId);
  }
  stopped = true;
}

/* addEventListener 监听事件 load 完成后执行 */
window.addEventListener("load", function() {
  var nCusped = document.getElementById("Cusped").value;
  r = R / nCusped;
  t = 0;
  start();
}, false);

/* addEventListener 监听事件 input 改变后执行 */
Cusped.addEventListener("input", function() {
  stopAnim();
  ctx.clearRect(0, 0, cw, ch)   //清空矩形
  var nCusped = this.value;
  document.getElementById("n").innerHTML = nCusped;
  r = R / nCusped;
  t = 0;
  start();
}, false);

/* addEventListener 监听事件 canvas 被点击时 toggle 动画 */
c.addEventListener("click", function() {
  (stopped == true) ? start() : stopAnim();
}, false);







CSS 微信雷达扫描动画

因在移动端呈现 animation （动画）以及 linear-gradient （线性渐变）效果，所以首先要考虑性能问题。在CSS中使用 will-change 来优化 transform 在浏览器中的渲染是一个不错的选择，但只有高版本的 Chrome 和 Opera 支持该属性。

核心 CSS Code 雷达扇叶效果：

/* 雷达扇叶效果 */
.pointerAnim{
    will-change: transform;
    -webkit-transform-origin: 50% 100%;
        -ms-transform-origin: 50% 100%;
            transform-origin: 50% 100%;
    border-radius: 50% 50% 0 0 / 100% 100% 0 0;
    background-image: -webkit-linear-gradient(315deg, rgba(255, 255, 255, 0.8) 0%, rgba(0, 0, 0, 0.02) 70%, transparent 100%);
    background-image: linear-gradient(135deg, rgba(255, 255, 255, 0.8) 0%, rgba(0, 0, 0, 0.02) 70%, transparent 100%);
}

执行动画 jQuery Code:

$(".btn").click(function () {
    $(".pointer").addClass("pointerAnim");  //添加雷达旋转动画
    var timeout = 1000;  //每隔1000ms
    var index = 0;
    var addClass;
    addClass = setInterval(function(){
        if(index >= $(".user span").length){
            index = 0;
            addClass = clearInterval(addClass);  //结束循环
            $(".user span").removeClass("flip").addClass("shadow")
        }else {
            $(".user span").eq(index++).fadeIn(timeout).addClass("flip")
        }
    },timeout);
})

JavaScript 正则表达式检验中文姓名与手机电话号码

Posted on 2015-10-08   |   In JavaScript / JQuery

检验中文姓名正则：姓名是2-4字的汉字（不包含少数民族姓名），不允许出现先生，小姐等字符。
/(?!.*先生.*|.*小姐.*|.*男士.*|.*女士.*|.*太太.*)^([\u4e00-\u9fa5\ ]{2,4})$/

检验电话号码正则：不得以数字0开头的各号段手机电话号码，如国内增加新号段，需要修改下面正则。
/^0?(13[0-9]|15[012356789]|17[678]|18[0-9]|14[57])[0-9]{8}$/

JavaScript Code：

window.onload = function () {
    var oName = document.getElementById('name');
    var oPhone = document.getElementById('phone');
    var oBtn = document.getElementById('btn');
    //检验姓名：姓名是2-4字的汉字
    var oIsName = /(?!.*先生.*|.*小姐.*|.*男士.*|.*女士.*|.*太太.*)^([\u4e00-\u9fa5\ ]{2,4})$/;
    //校验手机号码：必须以数字开头
    var oIsphone= /^0?(13[0-9]|15[012356789]|17[678]|18[0-9]|14[57])[0-9]{8}$/;

    oBtn.onclick = function () {
        //验证姓名不能为空
        if (oName.value=="") {
            alert("您还未输入姓名！");
        }
        //验证电话不能为空
        else if (oPhone.value=="") {
            alert(oName.value + "，您还未输入电话！");
        }
        //检验姓名
        else if(!oIsName.exec(oName.value)){
            alert(oName.value + "，请填写您的真实姓名！");
        }
        //校验手机号码
        else if(!oIsphone.exec(oPhone.value)){
            alert(oName.value + "，请填写您的真实手机号码！");
        }
    }
}
