JavaScript 封装 Ajax 读取TXT文件

Posted on 2015-08-26   |   In JavaScript / JQuery

如何使用 Ajax ：

  1. 创建ajax对象：XMLHttpRequest()
  2. 连接服务器：open()
  3. 发送请求：send()
  4. 接收服务器返回数据：onreadystatechange事件

未封装的 Ajax Code：

window.onload = function(){
    var oBtn = document.getElementById('btn');
    oBtn.onclick = function () {
        if (window.XMLHttpRequest) {  //if解决兼容IE6问题，window解决报错问题
            var oAjax = new XMLHttpRequest(); // 1.创建ajax对象,不兼容IE6
        }else{
            var oAjax = new ActiveXObject("Microsoft.XMLHTTP");  //兼容IE6
        };

        //open(方法，文件名，异步传输true)
        oAjax.open('GET','a.txt?t='+ new Date().getTime(),true);  //2.连接服务器
        //在文件名后添加动态时间是为了解决缓存问题 new Date().getTime()

        oAjax.send();    //3.发送请求

        //4.接收服务器返回数据
        oAjax.onreadystatechange = function (){    //使用onreadystatechange事件
            if (oAjax.readyState == 4) {    // 4表示解析（读取）完成，可以在客户端调用了
                if (oAjax.status == 200) {    // status 返回状态码,如果等于200，表示成功
                    alert("成功:" + oAjax.responseText);    //responseText 返回读取到的内容
                }else{
                    alert("失败:" + oAjax.status);    //如果失败使用status返回状态码，比如404等
                };
            };
        };
    };
};

封装后的 Ajax Code：

function ajax(url,fnSucc,fnFaild) {    //封装ajax
    if (window.XMLHttpRequest) {  //if解决兼容IE6问题，window解决报错问题
        var oAjax = new XMLHttpRequest(); // 1.创建ajax对象,不兼容IE6
    }else{
        var oAjax = new ActiveXObject("Microsoft.XMLHTTP");  //兼容IE6
    };

    //open(方法，文件名，异步传输true)
    oAjax.open('GET',url,true);  //2.连接服务器
    //在文件名后添加动态时间是为了解决缓存问题 new Date().getTime()

    oAjax.send();    //3.发送请求

    //4.接收服务器返回数据
    oAjax.onreadystatechange = function (){    //使用onreadystatechange事件
        if (oAjax.readyState == 4) {    // 4表示解析（读取）完成，可以在客户端调用了
            if (oAjax.status == 200) {    // status 返回状态码,如果等于200，表示成功

                fnSucc(oAjax.responseText)    //responseText 返回读取到的内容

            }else{

                if (fnFaild) {    //只有失败时才会调用if内的fnFaild(oAjax.status)
                    fnFaild(oAjax.status)    //如果失败使用status返回状态码，比如404等
                };
            };
        };
    };
};

Ajax 封装后的使用方法：

ajax('a.txt',function (str){    //使用封装好的ajax，记得调用封装好的 Ajax 文件
    alert(str);
})

JavaScript 实现可拖拽 DIV 效果

Posted on 2015-08-15   |   In JavaScript / JQuery
  1. 获取鼠标 onmousedown 的水平与垂直距离
  2. 当鼠标 onmousemove 时改变 DIV 的 left 与 top 值
  3. 当 onmouseup 后移除 onmousemove 和 自己本身 onmouseup

如果不想让 DIV 移出可视窗口，还可以进一步去优化。（详见 JavaScript Code）

JavaScript Code：

window.onload = function () {
    var oDiv = document.getElementById('divText')
    var disX = 0;
    var disY = 0;

    oDiv.onmousedown = function (ev){
        var oEvent = ev || event;
        disX = oEvent.clientX - oDiv.offsetLeft;  //获取水平距离
        disY = oEvent.clientY - oDiv.offsetTop;  //获取垂直距离

        document.onmousemove = function (ev){  //将鼠标事件对象设置为 document 防止移动速度过快而产生的 bug
            var oEvent = ev || event;

            var l = oEvent.clientX - disX;
            var t = oEvent.clientY - disY;

            //不让 div 移出可视窗口

            if (l < 0) {
                l = 0;  //不让 div 移出左侧可视窗口
            }else if (l > document.documentElement.clientWidth - oDiv.offsetWidth) {
                l = document.documentElement.clientWidth - oDiv.offsetWidth;  //不让 div 移出右侧可视窗口
            };

            if (t < 0) {
                t = 0;  //不让 div 移出顶部可视窗口
            }else if (t > document.documentElement.clientHeight - oDiv.offsetHeight) {
                t = document.documentElement.clientHeight - oDiv.offsetHeight;  //不让 div 移出底部可视窗口
            };

            //不让 div 移出可视窗口 End

            oDiv.style.left = l + 'px';  //当鼠标移动时改变div水平位置
            oDiv.style.top = t + 'px';  //当鼠标移动时改变div垂直位置
        }

        document.onmouseup = function (){  //当 onmouseup 后移除 onmousemove 和 自己本身 onmouseup
            document.onmousemove = null;
            document.onmouseup = null;
        }
    }
}

window.onload = function (){

        var oEvent = ev || event;

    }

JavaScript 倒计时功能代码分析

Posted on 2015-07-27   |   In JavaScript / JQuery

主要应用到的函数有：对数字进行舍余 Math.floor()，在N毫秒后执行函数 setTimeout("daojishi()",1000)

JavaScript 代码:

window.onload = function(){
    daojishi();
    }
    var countTime = 120*60;  //总秒数 7200 秒
    function daojishi(){
        if(countTime >= 0){
            var S = countTime%60;  //余数，每执行一次 countTime-- 后 S 将改变，所以需要使用 Math.floor
            var M = Math.floor(countTime/60);  //分钟  Math.floor 省略数字后的小数，取整数   M = 120
            if(M >= 60){
            var H = Math.floor(M/60);  //H = 2 小时
            M = Math.floor((countTime-H*60*60)/60);  // M = 0秒
            document.getElementById("timeVal").value = H +"时"+ M +"分"+ S + "秒";  //初始时间 2 时 0分 0秒
            }else if(M >= 1){
            document.getElementById("timeVal").value = M +"分"+ S + "秒";
            }else{
            document.getElementById("timeVal").value = S + "秒";
            }
            countTime--;
            vartt = window.setTimeout("daojishi()",1000);  //每秒执行一次 daojishi() 并 countTime--
        }else{
            window.clearTimeout(vartt);
        }
}

JavaScript 简易时钟效果实现方法

Posted on 2015-07-24   |   In JavaScript / JQuery

学习笔记：

Date() 时间对象，时 getHours() 分 getMinutes() 秒 getSeconds()

设置时间间隔 setInterval(function,1000) 每隔一秒执行一次函数，相似的还有 setTimeout(function,1000) 在1秒后执行函数，只执行一次。

如果需要关闭 setInterval() 可使用 clearInterval()

JavaScript 代码：

//给每个<10的数字前补0
function zero (n) {
    if (n<10) {
        return '0' + n;
    } else{
        return '' + n;
    };
}

window.onload = function (){
    function clock(){
        var oImg = document.getElementsByTagName('img');
        var oDate = new Date();
        var str = zero(oDate.getHours()) + zero(oDate.getMinutes()) + zero(oDate.getSeconds())  //获取时，分，秒后拼接成这样的字符串 var str = '015623'
        for (var i = 0; i < oImg.length; i++) {
                    oImg[i].src='images/'+str[i]+'.ico';
        };
    }
    setInterval(clock,1000)
    clock();  //在setInterval执行前运行一次，避免刷新归0
}
