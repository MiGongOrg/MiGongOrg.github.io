JavaScript 封装 Ajax 读取TXT文件

Posted on 2015-08-26   |   In JavaScript / JQuery

如何使用 Ajax ：

  1. 创建ajax对象：XMLHttpRequest()
  2. 连接服务器：open()
  3. 发送请求：send()
  4. 接收服务器返回数据：onreadystatechange事件

未封装的 Ajax Code：

window.onload = function(){
    var oBtn = document.getElementById('btn');
    oBtn.onclick = function () {
        if (window.XMLHttpRequest) {  //if解决兼容IE6问题，window解决报错问题
            var oAjax = new XMLHttpRequest(); // 1.创建ajax对象,不兼容IE6
        }else{
            var oAjax = new ActiveXObject("Microsoft.XMLHTTP");  //兼容IE6
        };

        //open(方法，文件名，异步传输true)
        oAjax.open('GET','a.txt?t='+ new Date().getTime(),true);  //2.连接服务器
        //在文件名后添加动态时间是为了解决缓存问题 new Date().getTime()

        oAjax.send();    //3.发送请求

        //4.接收服务器返回数据
        oAjax.onreadystatechange = function (){    //使用onreadystatechange事件
            if (oAjax.readyState == 4) {    // 4表示解析（读取）完成，可以在客户端调用了
                if (oAjax.status == 200) {    // status 返回状态码,如果等于200，表示成功
                    alert("成功:" + oAjax.responseText);    //responseText 返回读取到的内容
                }else{
                    alert("失败:" + oAjax.status);    //如果失败使用status返回状态码，比如404等
                };
            };
        };
    };
};

封装后的 Ajax Code：

function ajax(url,fnSucc,fnFaild) {    //封装ajax
    if (window.XMLHttpRequest) {  //if解决兼容IE6问题，window解决报错问题
        var oAjax = new XMLHttpRequest(); // 1.创建ajax对象,不兼容IE6
    }else{
        var oAjax = new ActiveXObject("Microsoft.XMLHTTP");  //兼容IE6
    };

    //open(方法，文件名，异步传输true)
    oAjax.open('GET',url,true);  //2.连接服务器
    //在文件名后添加动态时间是为了解决缓存问题 new Date().getTime()

    oAjax.send();    //3.发送请求

    //4.接收服务器返回数据
    oAjax.onreadystatechange = function (){    //使用onreadystatechange事件
        if (oAjax.readyState == 4) {    // 4表示解析（读取）完成，可以在客户端调用了
            if (oAjax.status == 200) {    // status 返回状态码,如果等于200，表示成功

                fnSucc(oAjax.responseText)    //responseText 返回读取到的内容

            }else{

                if (fnFaild) {    //只有失败时才会调用if内的fnFaild(oAjax.status)
                    fnFaild(oAjax.status)    //如果失败使用status返回状态码，比如404等
                };
            };
        };
    };
};

Ajax 封装后的使用方法：

ajax('a.txt',function (str){    //使用封装好的ajax，记得调用封装好的 Ajax 文件
    alert(str);
})
