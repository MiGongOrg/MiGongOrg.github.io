JavaScript 那些你不知道的小技巧

Posted on 2015-11-20   |   In JavaScript / JQuery

new Date() 时间转数字（时间戳）

//时间转数字（时间戳）
var oTime = + new Date();
console.log(oTime);  //1447925924936

String 转换数字，与 paresInt() 效果相同

//String 转换数字
var oString = + "123";
console.log(typeof oString);  //number

取整数的N种方法

var oNb1 = ~~3.1415926;     //取整数
console.log(oNb1);
var oNb2 = 3.1415926>>0;    //取整数
console.log(oNb2);
var oNb3 = 3.1415926^0;     //取整数
console.log(oNb3);
var oNb4 = 3.1415926|0;     //取整数
console.log(oNb4);
//所有取整数后均为 3

随机字符串

//8位，随机字符串
var oRandom1 = Math.random().toString(16).substring(2);
console.log(oRandom1);  //594a070a

//16位，随机字符串
var oRandom2 = Math.random().toString(36).substring(2);
console.log(oRandom2);  //pe65yv8dlxo5hfr

强制转换布尔类型

//转换布尔类型 true
var oBoolean = "a";
console.log(!!oBoolean);  //true

setTimeout 中获取其作用域外的变量

//setTimeout 中获取其作用域外的变量
for(var i = 0; i < 10; i++) {
    (function(i){
        setTimeout(function(){
            console.log(i);  //0-9
        },100);
    })(i)
}
