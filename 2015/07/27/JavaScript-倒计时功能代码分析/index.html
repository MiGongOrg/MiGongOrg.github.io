JavaScript 倒计时功能代码分析

Posted on 2015-07-27   |   In JavaScript / JQuery

主要应用到的函数有：对数字进行舍余 Math.floor()，在N毫秒后执行函数 setTimeout("daojishi()",1000)

JavaScript 代码:

window.onload = function(){
    daojishi();
    }
    var countTime = 120*60;  //总秒数 7200 秒
    function daojishi(){
        if(countTime >= 0){
            var S = countTime%60;  //余数，每执行一次 countTime-- 后 S 将改变，所以需要使用 Math.floor
            var M = Math.floor(countTime/60);  //分钟  Math.floor 省略数字后的小数，取整数   M = 120
            if(M >= 60){
            var H = Math.floor(M/60);  //H = 2 小时
            M = Math.floor((countTime-H*60*60)/60);  // M = 0秒
            document.getElementById("timeVal").value = H +"时"+ M +"分"+ S + "秒";  //初始时间 2 时 0分 0秒
            }else if(M >= 1){
            document.getElementById("timeVal").value = M +"分"+ S + "秒";
            }else{
            document.getElementById("timeVal").value = S + "秒";
            }
            countTime--;
            vartt = window.setTimeout("daojishi()",1000);  //每秒执行一次 daojishi() 并 countTime--
        }else{
            window.clearTimeout(vartt);
        }
}

HTML代码：

<input type="text" id="timeVal"/>

另一种比较简单的实现方法：（类似抢购商品倒计时）可设置截止时间

function GetRTime(){
    var EndTime= new Date('2025/09/17 19:00:00');   //设置截止时间
    var NowTime = new Date();   //获取当前时间
    var t =EndTime.getTime() - NowTime.getTime();   //截止时间 减去 当前时间
    var d=Math.floor(t/1000/60/60/24);
    var h=Math.floor(t/1000/60/60%24);
    var m=Math.floor(t/1000/60%60);
    var s=Math.floor(t/1000%60);

    document.getElementById("t_d").innerHTML = d + " 天";
    document.getElementById("t_h").innerHTML = h + " 时";
    document.getElementById("t_m").innerHTML = m + " 分";
    document.getElementById("t_s").innerHTML = s + " 秒";
}
setInterval(GetRTime,0);
